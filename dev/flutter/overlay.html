class _TextFieldWithSuggestionState extends State<TextFieldWithSuggestion> {
  FocusNode _focusNode = FocusNode(); // FocusNode ‡πÄ‡∏≠‡∏≤‡πÑ‡∏ß‡πâ‡∏Ñ‡∏ß‡∏ö‡∏Ñ‡∏∏‡∏°‡∏Å‡∏≤‡∏£‡πÇ‡∏ü‡∏Å‡∏±‡∏™ Widget

  @override
  void initState() {
    _focusNode.addListener(() {
      if (_focusNode.hasFocus) {
        _overlay = _createSuggestionOverlay();
        Overlay.of(context).insert(_overlay);
      } else {
        _overlay.remove();
      }
    });
    super.initState();
  }

  OverlayEntry _createSuggestionOverlay() {
    RenderBox renderBox =
        context.findRenderObject(); // ‡∏û‡∏¥‡∏Å‡∏±‡∏î ‡πÅ‡∏•‡∏∞‡∏Ç‡∏ô‡∏≤‡∏î‡∏Å‡∏≤‡∏£ Render ‡∏Ç‡∏≠‡∏á Widget ‡∏ô‡∏µ‡πâ

      builder: (context) => Positioned(
        left: offset.dx,
        top: offset.dy + size.height + 5.0,
        width: size.width,
        child: Material(
          elevation: 2.0,
          child: _suggestionList(),
        ),
      ),
    );
  }

  Widget _suggestionList() {

      height: 100.0,
      color: Colors.white,
      child: ListView(
        children: <Widget>[
          ListTile(title: Text('This')),
          ListTile(title: Text('is')),
          ListTile(title: Text('very')),
          ListTile(title: Text('awesome')),
        ],
      ),
    );
  }

  @override

    return TextField(
      decoration: InputDecoration(
        filled: true,
        fillColor: Colors.white,
        focusColor: Colors.white,
        border: OutlineInputBorder(
          borderRadius: BorderRadius.circular(5.0),
        ),
      ),
      focusNode: _focusNode,
    );
